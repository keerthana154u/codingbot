#include <stdio.h>

int automorphic(int num)
{
    int sqr=num*num;
    while(num!=0)
    {
        if(num%10!=sqr%10)
        {
            return 0;
        }
        num=num/10;
        sqr=sqr/10;
    }
    return 1;
}
int main()
{
    int num;
    printf("enter a number:");
    scanf("%d",&num);
    int sqr=num*num;
    if(automorphic(num))
    {
        printf("\nnum:%d\tsquare:%d\nThus automorphic number",num,sqr);
    }
    else
    {
        printf("num:%d\tsquare:%d\nThus not a automorphic number",num,sqr);
    }
}
