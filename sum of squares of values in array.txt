#include <stdio.h>

int sum_of_squares(int *arr,int n)
{
    int sum=0,sqr;
    for(int i=0;i<n;i++)
    {
        sqr=arr[i]*arr[i];
        sum=sum+sqr;
    }
    return sum;
}
int main()
{
    int arr[50];
    int n;
    printf("Enter array size:");
    scanf("%d",&n);
    printf("\nEnter array values:\n");
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("sum of squares of array values:%d",sum_of_squares(arr,n));
}
