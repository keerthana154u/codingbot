#include <stdio.h>
#include<stdio.h>
int main()
{
    char str[100];int i,j,flag=0;
    printf("Enter a statement:");
    gets(str);
    for(i='a',j='A';i<='z',j<='Z';i++,j++)
    {
        flag=0;
        for(int k=0;str[k]!='\0';k++)
        {
            if(str[k]==i||str[k]==j)
            {
                flag=1;
                break;
            }
        }
        if(flag==0)
            break;
    }
    if(flag==1)
        printf("Given statement is PANGRAM");
    else
        printf("Given statement is not a PANGRAM");
    return 0;
    
}
